// Fallback to basic attack on lowest-HP enemy if skill target is invalid/dead

  chosenAction = "basic";
  const aliveEnemies = units.filter(u => u.side !== "Hero" && u.isAlive);
  if (aliveEnemies.length > 0) {
    const sorted = [...aliveEnemies].sort((a, b) => a.hp - b.hp);
    targetId = sorted[0].id;
    validTarget = true;
  }
}
